// 다음 코드에서 해당하는 부분 찾아보기
int k;
for (k=10; k>=3; k--) {}
System.out.println("k : " + k);
// * 초기식: k = 10
// * 조건식: k >= 3
// * 증감식: k--
// * 출력 결과: k : 2

// ----------------------------------
// 다음 코드의 출력 결과를 예상해보기
for (int i=0; i<10; i+=2) {
    System.out.println("i : " + i);
}
// * 출력 결과:
i : 0
i : 2
i : 4
i : 6
i : 8
